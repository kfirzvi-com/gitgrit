check_readme.py 
def evaluate(project):
    files = project.list_files()
    readme = [f for f in files
              if f.lower().startswith("readme")]

    if readme:
        return {
            "passed": True,
            "score": 100,
            "message": f"Found {readme[0]}"
        }
